select * from (select O.RESPONSEDATE, O.ORDERREFERENCE,O.RESPONSEID,COUNT(I.ORDERRESPONSEITEMDID) from ORTORDERRESPONSE O left join ORTORDERRESPONSEITEM I
on (O.ORDERDID=I.ORDERDID
AND O.RESPONSEDATE in TO_DATE ('2014/09/19 00:00:00','YYYY/MM/DD HH24:MI:SS'))
GROUP BY I.ORDERRESPONSEITEMDID,O.RESPONSEID,O.ORDERREFERENCE,O.RESPONSEDATE
ORDER BY COUNT(I.ORDERRESPONSEITEMDID) desc) where RESPONSEDATE in TO_DATE ('2014/09/19 00:00:00','YYYY/MM/DD HH24:MI:SS');
